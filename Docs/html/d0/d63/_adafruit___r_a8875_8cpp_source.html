178 void Adafruit_RA8875::initialize(void) {
179  PLLinit();
180  writeReg(RA8875_SYSR, RA8875_SYSR_16BPP | RA8875_SYSR_MCU8);
181 
182  /* Timing values */
183  uint8_t pixclk;
184  uint8_t hsync_start;
185  uint8_t hsync_pw;
186  uint8_t hsync_finetune;
187  uint8_t hsync_nondisp;
188  uint8_t vsync_pw;
189  uint16_t vsync_nondisp;
190  uint16_t vsync_start;
191 
192  /* Set the correct values for the display being used */
193  if (_size == RA8875_480x272)
194  {
195  pixclk = RA8875_PCSR_PDATL | RA8875_PCSR_4CLK;
196  hsync_nondisp = 10;
197  hsync_start = 8;
198  hsync_pw = 48;
199  hsync_finetune = 0;
200  vsync_nondisp = 3;
201  vsync_start = 8;
202  vsync_pw = 10;
203  }
204  else // (_size == RA8875_800x480)
205  {
206  pixclk = RA8875_PCSR_PDATL | RA8875_PCSR_2CLK;
207  hsync_nondisp = 26;
208  hsync_start = 32;
209  hsync_pw = 96;
210  hsync_finetune = 0;
211  vsync_nondisp = 32;
212  vsync_start = 23;
213  vsync_pw = 2;
214  }
215 
216  writeReg(RA8875_PCSR, pixclk);
217  delay(1);
218 
219  /* Horizontal settings registers */
220  writeReg(RA8875_HDWR, (_width / 8) - 1); // H width: (HDWR + 1) * 8 = 480
221  writeReg(RA8875_HNDFTR, RA8875_HNDFTR_DE_HIGH + hsync_finetune);
222  writeReg(RA8875_HNDR, (hsync_nondisp - hsync_finetune - 2)/8); // H non-display: HNDR * 8 + HNDFTR + 2 = 10
223  writeReg(RA8875_HSTR, hsync_start/8 - 1); // Hsync start: (HSTR + 1)*8
224  writeReg(RA8875_HPWR, RA8875_HPWR_LOW + (hsync_pw/8 - 1)); // HSync pulse width = (HPWR+1) * 8
225 
226  /* Vertical settings registers */
227  writeReg(RA8875_VDHR0, (uint16_t)(_height - 1) & 0xFF);
228  writeReg(RA8875_VDHR1, (uint16_t)(_height - 1) >> 8);
229  writeReg(RA8875_VNDR0, vsync_nondisp-1); // V non-display period = VNDR + 1
230  writeReg(RA8875_VNDR1, vsync_nondisp >> 8);
231  writeReg(RA8875_VSTR0, vsync_start-1); // Vsync start position = VSTR + 1
232  writeReg(RA8875_VSTR1, vsync_start >> 8);
233  writeReg(RA8875_VPWR, RA8875_VPWR_LOW + vsync_pw - 1); // Vsync pulse width = VPWR + 1
234 
235  /* Set active window X */
236  writeReg(RA8875_HSAW0, 0); // horizontal start point
237  writeReg(RA8875_HSAW1, 0);
238  writeReg(RA8875_HEAW0, (uint16_t)(_width - 1) & 0xFF); // horizontal end point
239  writeReg(RA8875_HEAW1, (uint16_t)(_width - 1) >> 8);
240 
241  /* Set active window Y */
242  writeReg(RA8875_VSAW0, 0); // vertical start point
243  writeReg(RA8875_VSAW1, 0);
244  writeReg(RA8875_VEAW0, (uint16_t)(_height - 1) & 0xFF); // horizontal end point
245  writeReg(RA8875_VEAW1, (uint16_t)(_height - 1) >> 8);
246 
247  /* ToDo: Setup touch panel? */
248 
249  /* Clear the entire window */
250  writeReg(RA8875_MCLR, RA8875_MCLR_START | RA8875_MCLR_FULL);
251  delay(500);
252 }
