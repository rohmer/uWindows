210 //-----------------------------------------------------------------------------
211 static bool cardCommand(uint32_t xfertyp, uint32_t arg) {
212  DBG_IRQSTAT();
213  if (waitTimeout(isBusyCommandInhibit)) {
214  return false; // Caller will set errorCode.
215  }
216  SDHC_CMDARG = arg;
217  SDHC_XFERTYP = xfertyp;
218  if (waitTimeout(isBusyCommandComplete)) {
219  return false; // Caller will set errorCode.
220  }
221  m_irqstat = SDHC_IRQSTAT;
222  SDHC_IRQSTAT = m_irqstat;
223 
224  return (m_irqstat & SDHC_IRQSTAT_CC) &&
225  !(m_irqstat & SDHC_IRQSTAT_CMD_ERROR);
226 }
227 //-----------------------------------------------------------------------------
228 static bool cardCMD6(uint32_t arg, uint8_t* status) {
229  // CMD6 returns 64 bytes.
230  if (waitTimeout(isBusyCMD13)) {
231  return sdError(SD_CARD_ERROR_CMD13);
232  }
233  enableDmaIrs();
234  SDHC_DSADDR = (uint32_t)status;
235  SDHC_CMDARG = arg;
236  SDHC_BLKATTR = SDHC_BLKATTR_BLKCNT(1) | SDHC_BLKATTR_BLKSIZE(64);
237  SDHC_IRQSIGEN = SDHC_IRQSIGEN_MASK;
238  SDHC_XFERTYP = CMD6_XFERTYP;
239 
240  if (!waitDmaStatus()) {
241  return sdError(SD_CARD_ERROR_CMD6);
242  }
243  return true;
244 }

246 static void enableGPIO(bool enable) {
247  const uint32_t PORT_CLK = PORT_PCR_MUX(4) | PORT_PCR_DSE;
248  const uint32_t PORT_CMD_DATA = PORT_CLK | PORT_PCR_PS | PORT_PCR_PE;
249 
250  PORTE_PCR0 = enable ? PORT_CMD_DATA : 0; // SDHC_D1
251  PORTE_PCR1 = enable ? PORT_CMD_DATA : 0; // SDHC_D0
252  PORTE_PCR2 = enable ? PORT_CLK : 0; // SDHC_CLK
253  PORTE_PCR3 = enable ? PORT_CMD_DATA : 0; // SDHC_CMD
254  PORTE_PCR4 = enable ? PORT_CMD_DATA : 0; // SDHC_D3
255  PORTE_PCR5 = enable ? PORT_CMD_DATA : 0; // SDHC_D2
256 }

262 //-----------------------------------------------------------------------------
263 static void initSDHC() {
264 #ifdef HAS_KINETIS_MPU
265  // Allow SDHC Bus Master access.
266  MPU_RGDAAC0 |= 0x0C000000;
267 #endif
268  // Enable SDHC clock.
269  SIM_SCGC3 |= SIM_SCGC3_SDHC;
270 
271  // Disable GPIO clock.
272  enableGPIO(false);
273 
274  // Reset SDHC. Use default Water Mark Level of 16.
275  SDHC_SYSCTL = SDHC_SYSCTL_RSTA;
276  while (SDHC_SYSCTL & SDHC_SYSCTL_RSTA) {
277  }
278  // Set initial SCK rate.
279  setSdclk(400);
280 
281  enableGPIO(true);
282 
283  // Enable desired IRQSTAT bits.
284  SDHC_IRQSTATEN = SDHC_IRQSTATEN_MASK;
285 
286  NVIC_SET_PRIORITY(IRQ_SDHC, 6*16);
287  NVIC_ENABLE_IRQ(IRQ_SDHC);
288 
289  // Send 80 clocks to card.
290  SDHC_SYSCTL |= SDHC_SYSCTL_INITA;
291  while (SDHC_SYSCTL & SDHC_SYSCTL_INITA) {
292  }
293 }

326 //-----------------------------------------------------------------------------
327 static bool rdWrBlocks(uint32_t xfertyp,
328  uint32_t lba, uint8_t* buf, size_t n) {
329  if ((3 & (uint32_t)buf) || n == 0) {
330  return sdError(SD_CARD_ERROR_DMA);
331  }
332  if (yieldTimeout(isBusyCMD13)) {
333  return sdError(SD_CARD_ERROR_CMD13);
334  }
335  enableDmaIrs();
336  SDHC_DSADDR = (uint32_t)buf;
337  SDHC_CMDARG = m_highCapacity ? lba : 512*lba;
338  SDHC_BLKATTR = SDHC_BLKATTR_BLKCNT(n) | SDHC_BLKATTR_BLKSIZE(512);
339  SDHC_IRQSIGEN = SDHC_IRQSIGEN_MASK;
340  SDHC_XFERTYP = xfertyp;
341 
342  return waitDmaStatus();
343 }
344 //-----------------------------------------------------------------------------
345 // Read 16 byte CID or CSD register.
346 static bool readReg16(uint32_t xfertyp, void* data) {
347  uint8_t* d = reinterpret_cast<uint8_t*>(data);
348  if (!cardCommand(xfertyp, m_rca)) {
349  return false; // Caller will set errorCode.
350  }
351  uint32_t sr[] = {SDHC_CMDRSP0, SDHC_CMDRSP1, SDHC_CMDRSP2, SDHC_CMDRSP3};
352  for (int i = 0; i < 15; i++) {
353  d[14 - i] = sr[i/4] >> 8*(i%4);
354  }
355  d[15] = 0;
356  return true;
357 }
358 //-----------------------------------------------------------------------------
359 static void setSdclk(uint32_t kHzMax) {
360  const uint32_t DVS_LIMIT = 0X10;
361  const uint32_t SDCLKFS_LIMIT = 0X100;
362  uint32_t dvs = 1;
363  uint32_t sdclkfs = 1;
364  uint32_t maxSdclk = 1000*kHzMax;
365 
366  while ((F_CPU/(sdclkfs*DVS_LIMIT) > maxSdclk) && (sdclkfs < SDCLKFS_LIMIT)) {
367  sdclkfs <<= 1;
368  }
369  while ((F_CPU/(sdclkfs*dvs) > maxSdclk) && (dvs < DVS_LIMIT)) {
370  dvs++;
371  }
372  m_sdClkKhz = F_CPU/(1000*sdclkfs*dvs);
373  sdclkfs >>= 1;
374  dvs--;
375 
376  // Disable SDHC clock.
377  SDHC_SYSCTL &= ~SDHC_SYSCTL_SDCLKEN;
378 
379  // Change dividers.
380  uint32_t sysctl = SDHC_SYSCTL & ~(SDHC_SYSCTL_DTOCV_MASK
381  | SDHC_SYSCTL_DVS_MASK | SDHC_SYSCTL_SDCLKFS_MASK);
382 
383  SDHC_SYSCTL = sysctl | SDHC_SYSCTL_DTOCV(0x0E) | SDHC_SYSCTL_DVS(dvs)
384  | SDHC_SYSCTL_SDCLKFS(sdclkfs);
385 
386  // Wait until the SDHC clock is stable.
387  while (!(SDHC_PRSSTAT & SDHC_PRSSTAT_SDSTB)) {
388  }
389  // Enable the SDHC clock.
390  SDHC_SYSCTL |= SDHC_SYSCTL_SDCLKEN;
391 }
392 //-----------------------------------------------------------------------------
393 static bool transferStop() {
394  DBG_IRQSTAT();
395 
396  if (!cardCommand(CMD12_XFERTYP, 0)) {
397  return sdError(SD_CARD_ERROR_CMD12);
398  }
399  if (yieldTimeout(isBusyCMD13)) {
400  return sdError(SD_CARD_ERROR_CMD13);
401  }
402  // Save registers before reset DAT lines.
403  uint32_t irqsststen = SDHC_IRQSTATEN;
404  uint32_t proctl = SDHC_PROCTL & ~SDHC_PROCTL_SABGREQ;
405 
406  // Do reset to clear CDIHB. Should be a better way!
407  SDHC_SYSCTL |= SDHC_SYSCTL_RSTD;
408 
409  // Restore registers.
410  SDHC_IRQSTATEN = irqsststen;
411  SDHC_PROCTL = proctl;
412 
413  return true;
414 }
415 //-----------------------------------------------------------------------------
416 // Return true if timeout occurs.
417 static bool yieldTimeout(bool (*fcn)()) {
418  m_busyFcn = fcn;
419  uint32_t m = micros();
420  while (fcn()) {
421  if ((micros() - m) > BUSY_TIMEOUT_MICROS) {
422  m_busyFcn = 0;
423  return true;
424  }
425  yield();
426  }
427  m_busyFcn = 0;
428  return false; // Caller will set errorCode.
429 }
430 //-----------------------------------------------------------------------------
431 static bool waitDmaStatus() {
432  if (yieldTimeout(isBusyDMA)) {
433  return false; // Caller will set errorCode.
434  }
435  return (m_irqstat & SDHC_IRQSTAT_TC) && !(m_irqstat & SDHC_IRQSTAT_ERROR);
436 }
437 //-----------------------------------------------------------------------------
438 // Return true if timeout occurs.
439 static bool waitTimeout(bool (*fcn)()) {
440  uint32_t m = micros();
441  while (fcn()) {
442  if ((micros() - m) > BUSY_TIMEOUT_MICROS) {
443  return true;
444  }
445  }
446  return false; // Caller will set errorCode.
447 }

449 bool SdioCard::begin() {
450  uint32_t kHzSdClk;
451  uint32_t arg;
452  m_initDone = false;
453  m_errorCode = SD_CARD_ERROR_NONE;
454  m_highCapacity = false;
455  m_version2 = false;
456 
457  // initialize controller.
458  initSDHC();
459 
460  if (!cardCommand(CMD0_XFERTYP, 0)) {
461  return sdError(SD_CARD_ERROR_CMD0);
462  }
463  // Try several times for case of reset delay.
464  for (uint32_t i = 0; i < CMD8_RETRIES; i++) {
465  if (cardCommand(CMD8_XFERTYP, 0X1AA)) {
466  if (SDHC_CMDRSP0 != 0X1AA) {
467  return sdError(SD_CARD_ERROR_CMD8);
468  }
469  m_version2 = true;
470  break;
471  }
472  }
473  arg = m_version2 ? 0X40300000 : 0x00300000;
474  uint32_t m = micros();
475  do {
476  if (!cardAcmd(0, ACMD41_XFERTYP, arg) ||
477  ((micros() - m) > BUSY_TIMEOUT_MICROS)) {
478  return sdError(SD_CARD_ERROR_ACMD41);
479  }
480  } while ((SDHC_CMDRSP0 & 0x80000000) == 0);
481 
482  m_ocr = SDHC_CMDRSP0;
483  if (SDHC_CMDRSP0 & 0x40000000) {
484  // Is high capacity.
485  m_highCapacity = true;
486  }
487  if (!cardCommand(CMD2_XFERTYP, 0)) {
488  return sdError(SD_CARD_ERROR_CMD2);
489  }
490  if (!cardCommand(CMD3_XFERTYP, 0)) {
491  return sdError(SD_CARD_ERROR_CMD3);
492  }
493  m_rca = SDHC_CMDRSP0 & 0xFFFF0000;
494 
495  if (!readReg16(CMD9_XFERTYP, &m_csd)) {
496  return sdError(SD_CARD_ERROR_CMD9);
497  }
498  if (!readReg16(CMD10_XFERTYP, &m_cid)) {
499  return sdError(SD_CARD_ERROR_CMD10);
500  }
501  if (!cardCommand(CMD7_XFERTYP, m_rca)) {
502  return sdError(SD_CARD_ERROR_CMD7);
503  }
504  // Set card to bus width four.
505  if (!cardAcmd(m_rca, ACMD6_XFERTYP, 2)) {
506  return sdError(SD_CARD_ERROR_ACMD6);
507  }
508  // Set SDHC to bus width four.
509  SDHC_PROCTL &= ~SDHC_PROCTL_DTW_MASK;
510  SDHC_PROCTL |= SDHC_PROCTL_DTW(SDHC_PROCTL_DTW_4BIT);
511 
512  SDHC_WML = SDHC_WML_RDWML(FIFO_WML) | SDHC_WML_WRWML(FIFO_WML);
513 
514  // Determine if High Speed mode is supported and set frequency.
515  uint8_t status[64];
516  if (cardCMD6(0X00FFFFFF, status) && (2 & status[13]) &&
517  cardCMD6(0X80FFFFF1, status) && (status[16] & 0XF) == 1) {
518  kHzSdClk = 50000;
519  } else {
520  kHzSdClk = 25000;
521  }
522  // disable GPIO
523  enableGPIO(false);
524 
525  // Set the SDHC SCK frequency.
526  setSdclk(kHzSdClk);
527 
528  // enable GPIO
529  enableGPIO(true);
530  m_initDone = true;
531  return true;
532 }

537 //-----------------------------------------------------------------------------
538 bool SdioCard::erase(uint32_t firstBlock, uint32_t lastBlock) {
539  // check for single block erase
540  if (!m_csd.v1.erase_blk_en) {
541  // erase size mask
542  uint8_t m = (m_csd.v1.sector_size_high << 1) | m_csd.v1.sector_size_low;
543  if ((firstBlock & m) != 0 || ((lastBlock + 1) & m) != 0) {
544  // error card can't erase specified area
545  return sdError(SD_CARD_ERROR_ERASE_SINGLE_BLOCK);
546  }
547  }
548  if (!m_highCapacity) {

625 //-----------------------------------------------------------------------------
626 bool SdioCard::readData(uint8_t *dst) {
627  DBG_IRQSTAT();
628  uint32_t *p32 = reinterpret_cast<uint32_t*>(dst);
629 
630  if (!(SDHC_PRSSTAT & SDHC_PRSSTAT_RTA)) {
631  SDHC_PROCTL &= ~SDHC_PROCTL_SABGREQ;
632  if ((SDHC_BLKATTR & 0XFFFF0000) == 0X10000) {
633  // Don't stop at block gap if last block. Allows auto CMD12.
634  SDHC_PROCTL |= SDHC_PROCTL_CREQ;
635  } else {
636  noInterrupts();
637  SDHC_PROCTL |= SDHC_PROCTL_CREQ;
638  SDHC_PROCTL |= SDHC_PROCTL_SABGREQ;
639  interrupts();
640  }
641  }
642  if (waitTimeout(isBusyFifoRead)) {
643  return sdError(SD_CARD_ERROR_READ_FIFO);
644  }
645  for (uint32_t iw = 0 ; iw < 512/(4*FIFO_WML); iw++) {
646  while (0 == (SDHC_PRSSTAT & SDHC_PRSSTAT_BREN)) {
647  }
648  for (uint32_t i = 0; i < FIFO_WML; i++) {
649  p32[i] = SDHC_DATPORT;
650  }
651  p32 += FIFO_WML;
652  }
653  if (waitTimeout(isBusyTransferComplete)) {
654  return sdError(SD_CARD_ERROR_READ_TIMEOUT);
655  }
656  m_irqstat = SDHC_IRQSTAT;
657  SDHC_IRQSTAT = m_irqstat;
658  return (m_irqstat & SDHC_IRQSTAT_TC) && !(m_irqstat & SDHC_IRQSTAT_ERROR);
659 }
