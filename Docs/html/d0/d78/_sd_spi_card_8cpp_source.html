30 static uint8_t CRC7(const uint8_t* data, uint8_t n) {
31  uint8_t crc = 0;
32  for (uint8_t i = 0; i < n; i++) {
33  uint8_t d = data[i];
34  for (uint8_t j = 0; j < 8; j++) {
35  crc <<= 1;
36  if ((d & 0x80) ^ (crc & 0x80)) {
37  crc ^= 0x09;
38  }
39  d <<= 1;
40  }
41  }
42  return (crc << 1) | 1;
43 }

117 bool SdSpiCard::begin(SdSpiDriver* spi, uint8_t csPin, SPISettings settings) {
118  m_spiActive = false;
119  m_errorCode = SD_CARD_ERROR_NONE;
120  m_type = 0;
121  m_spiDriver = spi;
122  uint16_t t0 = curTimeMS();
123  uint32_t arg;
124 
125  m_spiDriver->begin(csPin);
126  m_spiDriver->setSpiSettings(SD_SCK_HZ(250000));
127  spiStart();
128 
129  // must supply min of 74 clock cycles with CS high.
130  spiUnselect();
131  for (uint8_t i = 0; i < 10; i++) {
132  spiSend(0XFF);
133  }
134  spiSelect();
135  // command to go idle in SPI mode
136  while (cardCommand(CMD0, 0) != R1_IDLE_STATE) {
137  if (isTimedOut(t0, SD_INIT_TIMEOUT)) {
138  error(SD_CARD_ERROR_CMD0);
139  goto fail;
140  }
141  }
142 #if USE_SD_CRC
143  if (cardCommand(CMD59, 1) != R1_IDLE_STATE) {
144  error(SD_CARD_ERROR_CMD59);
145  goto fail;
146  }
147 #endif // USE_SD_CRC
148  // check SD version
149  while (1) {
150  if (cardCommand(CMD8, 0x1AA) == (R1_ILLEGAL_COMMAND | R1_IDLE_STATE)) {
151  type(SD_CARD_TYPE_SD1);
152  break;
153  }
154  for (uint8_t i = 0; i < 4; i++) {
155  m_status = spiReceive();
156  }
157  if (m_status == 0XAA) {
158  type(SD_CARD_TYPE_SD2);
159  break;
160  }
161  if (isTimedOut(t0, SD_INIT_TIMEOUT)) {
162  error(SD_CARD_ERROR_CMD8);
163  goto fail;
164  }
165  }
166  // initialize card and send host supports SDHC if SD2
167  arg = type() == SD_CARD_TYPE_SD2 ? 0X40000000 : 0;
168 
169  while (cardAcmd(ACMD41, arg) != R1_READY_STATE) {
170  // check for timeout
171  if (isTimedOut(t0, SD_INIT_TIMEOUT)) {
172  error(SD_CARD_ERROR_ACMD41);
173  goto fail;
174  }
175  }
176  // if SD2 read OCR register to check for SDHC card
177  if (type() == SD_CARD_TYPE_SD2) {
178  if (cardCommand(CMD58, 0)) {
179  error(SD_CARD_ERROR_CMD58);
180  goto fail;
181  }
182  if ((spiReceive() & 0XC0) == 0XC0) {
183  type(SD_CARD_TYPE_SDHC);
184  }
185  // Discard rest of ocr - contains allowed voltage range.
186  for (uint8_t i = 0; i < 3; i++) {
187  spiReceive();
188  }
189  }
190  spiStop();
191  m_spiDriver->setSpiSettings(settings);
192  return true;
193 
194 fail:
195  spiStop();
196  return false;
197 }

199 // send command and return error code. Return zero for OK
200 uint8_t SdSpiCard::cardCommand(uint8_t cmd, uint32_t arg) {
201  // select card
202  if (!m_spiActive) {
203  spiStart();
204  }
205  // wait if busy
206  waitNotBusy(SD_WRITE_TIMEOUT);
207 
208 #if USE_SD_CRC
209  // form message
210  uint8_t buf[6];
211  buf[0] = (uint8_t)0x40U | cmd;
212  buf[1] = (uint8_t)(arg >> 24U);
213  buf[2] = (uint8_t)(arg >> 16U);
214  buf[3] = (uint8_t)(arg >> 8U);
215  buf[4] = (uint8_t)arg;
216 
217  // add CRC
218  buf[5] = CRC7(buf, 5);
219 
220  // send message
221  spiSend(buf, 6);
222 #else // USE_SD_CRC
223  // send command
224  spiSend(cmd | 0x40);
225 
226  // send argument
227  uint8_t *pa = reinterpret_cast<uint8_t *>(&arg);
228  for (int8_t i = 3; i >= 0; i--) {
229  spiSend(pa[i]);
230  }
231 
232  // send CRC - correct for CMD0 with arg zero or CMD8 with arg 0X1AA
233  spiSend(cmd == CMD0 ? 0X95 : 0X87);
234 #endif // USE_SD_CRC
235 
236  // skip stuff byte for stop read
237  if (cmd == CMD12) {
238  spiReceive();
239  }
240 
241  // wait for response
242  for (uint8_t i = 0; ((m_status = spiReceive()) & 0X80) && i != 0XFF; i++) {
243  }
244  return m_status;
245 }

252 bool SdSpiCard::erase(uint32_t firstBlock, uint32_t lastBlock) {
253  csd_t csd;
254  if (!readCSD(&csd)) {
255  goto fail;
256  }
257  // check for single block erase
258  if (!csd.v1.erase_blk_en) {
259  // erase size mask
260  uint8_t m = (csd.v1.sector_size_high << 1) | csd.v1.sector_size_low;
261  if ((firstBlock & m) != 0 || ((lastBlock + 1) & m) != 0) {
262  // error card can't erase specified area
263  error(SD_CARD_ERROR_ERASE_SINGLE_BLOCK);
264  goto fail;
265  }
266  }
267  if (m_type != SD_CARD_TYPE_SDHC) {
268  firstBlock <<= 9;
269  lastBlock <<= 9;
270  }
271  if (cardCommand(CMD32, firstBlock)
272  || cardCommand(CMD33, lastBlock)
273  || cardCommand(CMD38, 0)) {
274  error(SD_CARD_ERROR_ERASE);
275  goto fail;
276  }
277  if (!waitNotBusy(SD_ERASE_TIMEOUT)) {
278  error(SD_CARD_ERROR_ERASE_TIMEOUT);
279  goto fail;
280  }
281  spiStop();
282  return true;
283 
284 fail:
285  spiStop();
286  return false;
287 }

294 bool SdSpiCard::isBusy() {
295  bool rtn = true;
296  bool spiActive = m_spiActive;
297  if (!spiActive) {
298  spiStart();
299  }
300  for (uint8_t i = 0; i < 8; i++) {
301  if (0XFF == spiReceive()) {
302  rtn = false;
303  break;
304  }
305  }
306  if (!spiActive) {
307  spiStop();
308  }
309  return rtn;
310 }

312 bool SdSpiCard::isTimedOut(uint16_t startMS, uint16_t timeoutMS) {
313 #if WDT_YIELD_TIME_MICROS
314  static uint32_t last;
315  if ((micros() - last) > WDT_YIELD_TIME_MICROS) {
316  SysCall::yield();
317  last = micros();
318  }
319 #endif // WDT_YIELD_TIME_MICROS
320  return (curTimeMS() - startMS) > timeoutMS;
321 }

359 //------------------------------------------------------------------------------
360 bool SdSpiCard::readData(uint8_t* dst, size_t count) {
361 #if USE_SD_CRC
362  uint16_t crc;
363 #endif // USE_SD_CRC
364  // wait for start block token
365  uint16_t t0 = curTimeMS();
366  while ((m_status = spiReceive()) == 0XFF) {
367  if (isTimedOut(t0, SD_READ_TIMEOUT)) {
368  error(SD_CARD_ERROR_READ_TIMEOUT);
369  goto fail;
370  }
371  }
372  if (m_status != DATA_START_BLOCK) {
373  error(SD_CARD_ERROR_READ);
374  goto fail;
375  }
376  // transfer data
377  if ((m_status = spiReceive(dst, count))) {
378  error(SD_CARD_ERROR_DMA);
379  goto fail;
380  }
381 
382 #if USE_SD_CRC
383  // get crc
384  crc = (spiReceive() << 8) | spiReceive();
385  if (crc != CRC_CCITT(dst, count)) {
386  error(SD_CARD_ERROR_READ_CRC);
387  goto fail;
388  }
389 #else
390  // discard crc
391  spiReceive();
392  spiReceive();
393 #endif // USE_SD_CRC
394  return true;
395 
396 fail:
397  spiStop();
398  return false;
399 }

501 // wait for card to go not busy
502 bool SdSpiCard::waitNotBusy(uint16_t timeoutMS) {
503  uint16_t t0 = curTimeMS();
504 #if WDT_YIELD_TIME_MICROS
505  // Call isTimedOut first to insure yield is called.
506  while (!isTimedOut(t0, timeoutMS)) {
507  if (spiReceive() == 0XFF) {
508  return true;
509  }
510  }
511  return false;
512 #else // WDT_YIELD_TIME_MICROS
513  // Check not busy first since yield is not called in isTimedOut.
514  while (spiReceive() != 0XFF) {
515  if (isTimedOut(t0, timeoutMS)) {
516  return false;
517  }
518  }
519  return true;
520 #endif // WDT_YIELD_TIME_MICROS
521 }

591 // send one block of data for write block or write multiple blocks
592 bool SdSpiCard::writeData(uint8_t token, const uint8_t* src) {
593 #if USE_SD_CRC
594  uint16_t crc = CRC_CCITT(src, 512);
595 #else // USE_SD_CRC
596  uint16_t crc = 0XFFFF;
597 #endif // USE_SD_CRC
598  spiSend(token);
599  spiSend(src, 512);
600  spiSend(crc >> 8);
601  spiSend(crc & 0XFF);
602 
603  m_status = spiReceive();
604  if ((m_status & DATA_RES_MASK) != DATA_RES_ACCEPTED) {
605  error(SD_CARD_ERROR_WRITE);
606  goto fail;
607  }
608  return true;
609 
610 fail:
611  spiStop();
612  return false;
