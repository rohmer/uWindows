757 static inline uint16_t FAT_DATE(uint16_t year, uint8_t month, uint8_t day) {
758  return (year - 1980) << 9 | month << 5 | day;
759 }
765 static inline uint16_t FAT_YEAR(uint16_t fatDate) {
766  return 1980 + (fatDate >> 9);
767 }
773 static inline uint8_t FAT_MONTH(uint16_t fatDate) {
774  return (fatDate >> 5) & 0XF;
775 }
781 static inline uint8_t FAT_DAY(uint16_t fatDate) {
782  return fatDate & 0X1F;
783 }
791 static inline uint16_t FAT_TIME(uint8_t hour, uint8_t minute, uint8_t second) {
792  return hour << 11 | minute << 5 | second >> 1;
793 }
799 static inline uint8_t FAT_HOUR(uint16_t fatTime) {
800  return fatTime >> 11;
801 }
807 static inline uint8_t FAT_MINUTE(uint16_t fatTime) {
808  return (fatTime >> 5) & 0X3F;
809 }
817 static inline uint8_t FAT_SECOND(uint16_t fatTime) {
818  return 2*(fatTime & 0X1F);
819 }
