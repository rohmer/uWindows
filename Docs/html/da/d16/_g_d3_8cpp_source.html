570  uint16_t a;

572 
573  /* These values are tricky. So pretend they are not */
574  if (x == -32768)
575  x++;
576  if (y == -32768)
577  y++;
578 
579  if ((x <= 0) ^ (y > 0)) {
580  int16_t t; t = x; x = y; y = t;
581  xx ^= 0x4000;
582  }
583  if (x <= 0) {
584  x = -x;
585  } else {
586  xx ^= 0x8000;
587  }
588  y = abs(y);
589  if (x > y) {
590  int16_t t; t = x; x = y; y = t;
591  xx ^= 0x3fff;
592  }
593  while ((x | y) & 0xff80) {
594  x >>= 1;

596  }

599  } else if (x == y) {

605  // assert(r < 256);
606  a = pgm_read_byte(atan8 + r) << 5;
