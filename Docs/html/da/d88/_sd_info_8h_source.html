457 //-----------------------------------------------------------------------------
458 inline uint32_t sdCardCapacity(csd_t* csd) {
459  if (csd->v1.csd_ver == 0) {
460  uint8_t read_bl_len = csd->v1.read_bl_len;
461  uint16_t c_size = (csd->v1.c_size_high << 10)
462  | (csd->v1.c_size_mid << 2) | csd->v1.c_size_low;
463  uint8_t c_size_mult = (csd->v1.c_size_mult_high << 1)
464  | csd->v1.c_size_mult_low;
465  return (uint32_t)(c_size + 1) << (c_size_mult + read_bl_len - 7);
466  } else if (csd->v2.csd_ver == 1) {
467  uint32_t c_size = 0X10000L * csd->v2.c_size_high + 0X100L
468  * (uint32_t)csd->v2.c_size_mid + csd->v2.c_size_low;
469  return (c_size + 1) << 10;
470  } else {
471  return 0;
472  }
473 }
