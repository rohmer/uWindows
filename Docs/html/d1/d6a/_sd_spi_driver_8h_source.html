318 inline uint8_t SdSpiAltDriver::receive(uint8_t* buf, size_t n) {
319  if (n-- == 0) {
320  return 0;
321  }
322  SPDR = 0XFF;
323  for (size_t i = 0; i < n; i++) {
324  while (!(SPSR & (1 << SPIF))) {}
325  uint8_t b = SPDR;
326  SPDR = 0XFF;
327  buf[i] = b;
328  }
329  while (!(SPSR & (1 << SPIF))) {}
330  buf[n] = SPDR;
331  return 0;
332 }

337 }

339 inline void SdSpiAltDriver::send(const uint8_t* buf , size_t n) {
340  if (n == 0) {
341  return;
342  }
343  SPDR = buf[0];
344  if (n > 1) {
345  uint8_t b = buf[1];
346  size_t i = 2;
347  while (1) {
348  while (!(SPSR & (1 << SPIF))) {}
349  SPDR = b;
350  if (i == n) {
351  break;
352  }
353  b = buf[i++];
354  }
355  }
356  while (!(SPSR & (1 << SPIF))) {}
357 }
