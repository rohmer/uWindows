21 bool SdSpiCardEX::readBlock(uint32_t block, uint8_t* dst) {
22  if (m_curState != READ_STATE || block != m_curBlock) {
23  if (!syncBlocks()) {
24  return false;
25  }
26  if (!SdSpiCard::readStart(block)) {
27  return false;
28  }
29  m_curBlock = block;
30  m_curState = READ_STATE;
31  }
32  if (!SdSpiCard::readData(dst)) {
33  return false;
34  }
35  m_curBlock++;
36  return true;
37 }
