23 int16_t FatStreamBase::getch() {
24  uint8_t c;
25  int8_t s = read(&c, 1);
26  if (s != 1) {
27  if (s < 0) {
28  setstate(badbit);
29  } else {
30  setstate(eofbit);
31  }
32  return -1;
33  }
34  if (c != '\r' || (getmode() & ios::binary)) {
35  return c;
36  }
37  s = read(&c, 1);
38  if (s == 1 && c == '\n') {
39  return c;
40  }
41  if (s == 1) {
42  seekCur(-1);
43  }
44  return '\r';
45 }
