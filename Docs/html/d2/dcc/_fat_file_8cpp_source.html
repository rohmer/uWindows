33 //------------------------------------------------------------------------------
34 // Add a cluster to a directory file and zero the cluster.
35 // Return with first block of cluster in the cache.
36 bool FatFile::addDirCluster() {
37  uint32_t block;
38  cache_t* pc;
39 
40  if (isRootFixed()) {
41  DBG_FAIL_MACRO;
42  goto fail;
43  }
44  // max folder size
45  if (m_curPosition >= 512UL*4095) {
46  DBG_FAIL_MACRO;
47  goto fail;
48  }
49  if (!addCluster()) {
50  DBG_FAIL_MACRO;
51  goto fail;
52  }
53  block = m_vol->clusterFirstBlock(m_curCluster);
54  pc = m_vol->cacheFetchData(block, FatCache::CACHE_RESERVE_FOR_WRITE);
55  if (!pc) {
56  DBG_FAIL_MACRO;
57  goto fail;
58  }
59  memset(pc, 0, 512);
60  // zero rest of clusters
61  for (uint8_t i = 1; i < m_vol->blocksPerCluster(); i++) {
62  if (!m_vol->writeBlock(block + i, pc->data)) {
63  DBG_FAIL_MACRO;
64  goto fail;
65  }
66  }
67  // Set position to EOF to avoid inconsistent curCluster/curPosition.
68  m_curPosition += 512UL*m_vol->blocksPerCluster();
69  return true;
70 
71 fail:
72  return false;
73 }

96 bool FatFile::contiguousRange(uint32_t* bgnBlock, uint32_t* endBlock) {
97  // error if no blocks
98  if (m_firstCluster == 0) {
99  DBG_FAIL_MACRO;
100  goto fail;
101  }
102  for (uint32_t c = m_firstCluster; ; c++) {
103  uint32_t next;
104  int8_t fg = m_vol->fatGet(c, &next);
105  if (fg < 0) {
106  DBG_FAIL_MACRO;
107  goto fail;
108  }
109  // check for contiguous
110  if (fg == 0 || next != (c + 1)) {
111  // error if not end of chain
112  if (fg) {
113  DBG_FAIL_MACRO;
114  goto fail;
115  }
116  *bgnBlock = m_vol->clusterFirstBlock(m_firstCluster);
117  *endBlock = m_vol->clusterFirstBlock(c)
118  + m_vol->blocksPerCluster() - 1;
119  return true;
120  }
121  }
122 
123 fail:
124  return false;
125 }

180 uint8_t FatFile::dirName(const dir_t* dir, char* name) {
181  uint8_t j = 0;
182  uint8_t lcBit = DIR_NT_LC_BASE;
183  for (uint8_t i = 0; i < 11; i++) {
184  if (dir->name[i] == ' ') {
185  continue;
186  }
187  if (i == 8) {
188  // Position bit for extension.
189  lcBit = DIR_NT_LC_EXT;
190  name[j++] = '.';
191  }
192  char c = dir->name[i];
193  if ('A' <= c && c <= 'Z' && (lcBit & dir->reservedNT)) {
194  c += 'a' - 'A';
195  }
196  name[j++] = c;
197  }
198  name[j] = 0;
199  return j;
200 }

202 //------------------------------------------------------------------------------
203 uint32_t FatFile::dirSize() {
204  int8_t fg;
205  if (!isDir()) {
206  return 0;
207  }
208  if (isRootFixed()) {
209  return 32*m_vol->rootDirEntryCount();
210  }
211  uint16_t n = 0;
212  uint32_t c = isRoot32() ? m_vol->rootDirStart() : m_firstCluster;
213  do {
214  fg = m_vol->fatGet(c, &c);
215  if (fg < 0 || n > 4095) {
216  return 0;
217  }
218  n += m_vol->blocksPerCluster();
219  } while (fg);
220  return 512UL*n;
221 }

223 int16_t FatFile::fgets(char* str, int16_t num, char* delim) {
224  char ch;
225  int16_t n = 0;
226  int16_t r = -1;
227  while ((n + 1) < num && (r = read(&ch, 1)) == 1) {
228  // delete CR
229  if (ch == '\r') {
230  continue;
231  }
232  str[n++] = ch;
233  if (!delim) {
234  if (ch == '\n') {
235  break;
236  }
237  } else {
238  if (strchr(delim, ch)) {
239  break;
240  }
241  }
242  }
243  if (r < 0) {
244  // read error
245  return -1;
246  }
247  str[n] = '\0';
248  return n;
249 }

298 bool FatFile::mkdir(FatFile* parent, fname_t* fname) {
299  uint32_t block;
300  dir_t dot;
301  dir_t* dir;
302  cache_t* pc;
303 
304  if (!parent->isDir()) {
305  DBG_FAIL_MACRO;
306  goto fail;
307  }
308  // create a normal file
309  if (!open(parent, fname, O_CREAT | O_EXCL | O_RDWR)) {
310  DBG_FAIL_MACRO;
311  goto fail;
312  }
313  // convert file to directory
314  m_flags = O_READ;
315  m_attr = FILE_ATTR_SUBDIR;
316 
317  // allocate and zero first cluster
318  if (!addDirCluster()) {
319  DBG_FAIL_MACRO;
320  goto fail;
321  }
322  m_firstCluster = m_curCluster;
323  // Set to start of dir
324  rewind();
325  // force entry to device
326  if (!sync()) {
327  DBG_FAIL_MACRO;
328  goto fail;
329  }
330  // cache entry - should already be in cache due to sync() call
331  dir = cacheDirEntry(FatCache::CACHE_FOR_WRITE);
332  if (!dir) {
333  DBG_FAIL_MACRO;
334  goto fail;
335  }
336  // change directory entry attribute
337  dir->attributes = DIR_ATT_DIRECTORY;
338 
339  // make entry for '.'
340  memcpy(&dot, dir, sizeof(dot));
341  dot.name[0] = '.';
342  for (uint8_t i = 1; i < 11; i++) {
343  dot.name[i] = ' ';
344  }
345 
346  // cache block for '.' and '..'
347  block = m_vol->clusterFirstBlock(m_firstCluster);
348  pc = m_vol->cacheFetchData(block, FatCache::CACHE_FOR_WRITE);
349  if (!pc) {
350  DBG_FAIL_MACRO;
351  goto fail;
352  }
353  // copy '.' to block
354  memcpy(&pc->dir[0], &dot, sizeof(dot));
355  // make entry for '..'
356  dot.name[1] = '.';
357  dot.firstClusterLow = parent->m_firstCluster & 0XFFFF;
358  dot.firstClusterHigh = parent->m_firstCluster >> 16;
359  // copy '..' to block
360  memcpy(&pc->dir[1], &dot, sizeof(dot));
361  // write first block
362  return m_vol->cacheSync();
363 
364 fail:
365  return false;
366 }

415 //------------------------------------------------------------------------------
416 bool FatFile::open(FatFile* dirFile, uint16_t index, uint8_t oflag) {
417  uint8_t chksum = 0;
418  uint8_t lfnOrd = 0;
419  dir_t* dir;
420  ldir_t*ldir;
421 
422  // Error if already open.
423  if (isOpen() || !dirFile->isDir()) {
424  DBG_FAIL_MACRO;
425  goto fail;
426  }
427  // Don't open existing file if O_EXCL - user call error.
428  if (oflag & O_EXCL) {
429  DBG_FAIL_MACRO;
430  goto fail;
431  }
432  if (index) {
433  // Check for LFN.
434  if (!dirFile->seekSet(32UL*(index -1))) {
435  DBG_FAIL_MACRO;
436  goto fail;
437  }
438  ldir = reinterpret_cast<ldir_t*>(dirFile->readDirCache());
439  if (!ldir) {
440  DBG_FAIL_MACRO;
441  goto fail;
442  }
443  if (ldir->attr == DIR_ATT_LONG_NAME) {
444  if (1 == (ldir->ord & 0X1F)) {
445  chksum = ldir->chksum;
446  // Use largest possible number.
447  lfnOrd = index > 20 ? 20 : index;
448  }
449  }
450  } else {
451  dirFile->rewind();
452  }
453  // read entry into cache
454  dir = dirFile->readDirCache();
455  if (!dir) {
456  DBG_FAIL_MACRO;
457  goto fail;
458  }
459  // error if empty slot or '.' or '..'
460  if (dir->name[0] == DIR_NAME_DELETED ||
461  dir->name[0] == DIR_NAME_FREE ||
462  dir->name[0] == '.') {
463  DBG_FAIL_MACRO;
464  goto fail;
465  }
466  if (lfnOrd && chksum != lfnChecksum(dir->name)) {
467  DBG_FAIL_MACRO;
468  goto fail;
469  }
470  // open cached entry
471  if (!openCachedEntry(dirFile, index, oflag, lfnOrd)) {
472  DBG_FAIL_MACRO;
473  goto fail;
474  }
475  return true;
476 
477 fail:
478  return false;
479 }

482 
483 bool FatFile::openCachedEntry(FatFile* dirFile, uint16_t dirIndex,
484  uint8_t oflag, uint8_t lfnOrd) {
485  uint32_t firstCluster;
486  memset(this, 0, sizeof(FatFile));
487  // location of entry in cache
488  m_vol = dirFile->m_vol;
489  m_dirIndex = dirIndex;
490  m_dirCluster = dirFile->m_firstCluster;
491  dir_t* dir = &m_vol->cacheAddress()->dir[0XF & dirIndex];
492 
493  // Must be file or subdirectory.
494  if (!DIR_IS_FILE_OR_SUBDIR(dir)) {
495  DBG_FAIL_MACRO;
496  goto fail;
497  }
498  m_attr = dir->attributes & FILE_ATTR_COPY;
499  if (DIR_IS_FILE(dir)) {
500  m_attr |= FILE_ATTR_FILE;
501  }
502  m_lfnOrd = lfnOrd;
503  // Write, truncate, or at end is an error for a directory or read-only file.
504  if (oflag & (O_WRITE | O_TRUNC | O_AT_END)) {
505  if (isSubDir() || isReadOnly()) {
506  DBG_FAIL_MACRO;
507  goto fail;
508  }
509  }
510  // save open flags for read/write
511  m_flags = oflag & F_OFLAG;
512 
513  m_dirBlock = m_vol->cacheBlockNumber();
514 
515  // copy first cluster number for directory fields
516  firstCluster = ((uint32_t)dir->firstClusterHigh << 16)
517  | dir->firstClusterLow;
518 
519  if (oflag & O_TRUNC) {
520  if (firstCluster && !m_vol->freeChain(firstCluster)) {
521  DBG_FAIL_MACRO;
522  goto fail;
523  }
524  // need to update directory entry
525  m_flags |= F_FILE_DIR_DIRTY;
526  } else {
527  m_firstCluster = firstCluster;
528  m_fileSize = dir->fileSize;
529  }
530  if ((oflag & O_AT_END) && !seekSet(m_fileSize)) {
531  DBG_FAIL_MACRO;
532  goto fail;
533  }
534  return true;
535 
536 fail:
537  m_attr = FILE_ATTR_CLOSED;
538  return false;
539 }

541 bool FatFile::openNext(FatFile* dirFile, uint8_t oflag) {
542  uint8_t chksum = 0;
543  ldir_t* ldir;
544  uint8_t lfnOrd = 0;
545  uint16_t index;
546 
547  // Check for not open and valid directory..
548  if (isOpen() || !dirFile->isDir() || (dirFile->curPosition() & 0X1F)) {
549  DBG_FAIL_MACRO;
550  goto fail;
551  }
552  while (1) {
553  // read entry into cache
554  index = dirFile->curPosition()/32;
555  dir_t* dir = dirFile->readDirCache();
556  if (!dir) {
557  if (dirFile->getError()) {
558  DBG_FAIL_MACRO;
559  }
560  goto fail;
561  }
562  // done if last entry
563  if (dir->name[0] == DIR_NAME_FREE) {
564  goto fail;
565  }
566  // skip empty slot or '.' or '..'
567  if (dir->name[0] == '.' || dir->name[0] == DIR_NAME_DELETED) {
568  lfnOrd = 0;
569  } else if (DIR_IS_FILE_OR_SUBDIR(dir)) {
570  if (lfnOrd && chksum != lfnChecksum(dir->name)) {
571  DBG_FAIL_MACRO;
572  goto fail;
573  }
574  if (!openCachedEntry(dirFile, index, oflag, lfnOrd)) {
575  DBG_FAIL_MACRO;
576  goto fail;
577  }
578  return true;
579  } else if (DIR_IS_LONG_NAME(dir)) {
580  ldir = reinterpret_cast<ldir_t*>(dir);
581  if (ldir->ord & LDIR_ORD_LAST_LONG_ENTRY) {
582  lfnOrd = ldir->ord & 0X1F;
583  chksum = ldir->chksum;
584  }
585  } else {
586  lfnOrd = 0;
587  }
588  }
589 
590 fail:
591  return false;
592 }

695 //------------------------------------------------------------------------------
696 int FatFile::read(void* buf, size_t nbyte) {
697  int8_t fg;
698  uint8_t blockOfCluster = 0;
699  uint8_t* dst = reinterpret_cast<uint8_t*>(buf);
700  uint16_t offset;
701  size_t toRead;
702  uint32_t block; // raw device block number
703  cache_t* pc;
704 
705  // error if not open for read
706  if (!isOpen() || !(m_flags & O_READ)) {
707  DBG_FAIL_MACRO;
708  goto fail;
709  }
710 
711  if (isFile()) {
712  uint32_t tmp32 = m_fileSize - m_curPosition;
713  if (nbyte >= tmp32) {
714  nbyte = tmp32;
715  }
716  } else if (isRootFixed()) {
717  uint16_t tmp16 = 32*m_vol->m_rootDirEntryCount - (uint16_t)m_curPosition;
718  if (nbyte > tmp16) {
719  nbyte = tmp16;
720  }
721  }
722  toRead = nbyte;
723  while (toRead) {
724  size_t n;
725  offset = m_curPosition & 0X1FF; // offset in block
726  if (isRootFixed()) {
727  block = m_vol->rootDirStart() + (m_curPosition >> 9);
728  } else {
729  blockOfCluster = m_vol->blockOfCluster(m_curPosition);
730  if (offset == 0 && blockOfCluster == 0) {
731  // start of new cluster
732  if (m_curPosition == 0) {
733  // use first cluster in file
734  m_curCluster = isRoot32() ? m_vol->rootDirStart() : m_firstCluster;
735  } else {
736  // get next cluster from FAT
737  fg = m_vol->fatGet(m_curCluster, &m_curCluster);
738  if (fg < 0) {
739  DBG_FAIL_MACRO;
740  goto fail;
741  }
742  if (fg == 0) {
743  if (isDir()) {
744  break;
745  }
746  DBG_FAIL_MACRO;
747  goto fail;
748  }
749  }
750  }
751  block = m_vol->clusterFirstBlock(m_curCluster) + blockOfCluster;
752  }
753  if (offset != 0 || toRead < 512 || block == m_vol->cacheBlockNumber()) {
754  // amount to be read from current block
755  n = 512 - offset;
756  if (n > toRead) {
757  n = toRead;
758  }
759  // read block to cache and copy data to caller
760  pc = m_vol->cacheFetchData(block, FatCache::CACHE_FOR_READ);
761  if (!pc) {
762  DBG_FAIL_MACRO;
763  goto fail;
764  }
765  uint8_t* src = pc->data + offset;
766  memcpy(dst, src, n);
767 #if USE_MULTI_BLOCK_IO
768  } else if (toRead >= 1024) {
769  uint8_t nb = toRead >> 9;
770  if (!isRootFixed()) {
771  uint8_t mb = m_vol->blocksPerCluster() - blockOfCluster;
772  if (mb < nb) {
773  nb = mb;
774  }
775  }
776  n = 512*nb;
777  if (block <= m_vol->cacheBlockNumber()
778  && block < (m_vol->cacheBlockNumber() + nb)) {
779  // flush cache if a block is in the cache
780  if (!m_vol->cacheSyncData()) {
781  DBG_FAIL_MACRO;
782  goto fail;
783  }
784  }
785  if (!m_vol->readBlocks(block, dst, nb)) {
786  DBG_FAIL_MACRO;
787  goto fail;
788  }
789 #endif // USE_MULTI_BLOCK_IO
790  } else {
791  // read single block
792  n = 512;
793  if (!m_vol->readBlock(block, dst)) {
794  DBG_FAIL_MACRO;
795  goto fail;
796  }
797  }
798  dst += n;
799  m_curPosition += n;
800  toRead -= n;
801  }
802  return nbyte - toRead;
803 
804 fail:
805  m_error |= READ_ERROR;
806  return -1;
807 }

872 //------------------------------------------------------------------------------
873 bool FatFile::rename(FatFile* dirFile, const char* newPath) {
874  dir_t entry;
875  uint32_t dirCluster = 0;
876  FatFile file;
877  FatFile oldFile;
878  cache_t* pc;
879  dir_t* dir;
880 
881  // Must be an open file or subdirectory.
882  if (!(isFile() || isSubDir())) {
883  DBG_FAIL_MACRO;
884  goto fail;
885  }
886  // Can't rename LFN in 8.3 mode.
887  if (!USE_LONG_FILE_NAMES && isLFN()) {
888  DBG_FAIL_MACRO;
889  goto fail;
890  }
891  // Can't move file to new volume.
892  if (m_vol != dirFile->m_vol) {
893  DBG_FAIL_MACRO;
894  goto fail;
895  }

897  sync();

900  if (!dir) {
901  DBG_FAIL_MACRO;
902  goto fail;
903  }
904  // save directory entry

906  // make directory entry for new path
907  if (isFile()) {
908  if (!file.open(dirFile, newPath, O_CREAT | O_EXCL | O_WRITE)) {
909  DBG_FAIL_MACRO;
910  goto fail;
911  }
912  } else {
913  // don't create missing path prefix components
914  if (!file.mkdir(dirFile, newPath, false)) {
915  DBG_FAIL_MACRO;
916  goto fail;
917  }
918  // save cluster containing new dot dot
919  dirCluster = file.m_firstCluster;
920  }
921  // change to new directory entry
922 
923  m_dirBlock = file.m_dirBlock;
924  m_dirIndex = file.m_dirIndex;
925  m_lfnOrd = file.m_lfnOrd;
926  m_dirCluster = file.m_dirCluster;
927  // mark closed to avoid possible destructor close call
928  file.m_attr = FILE_ATTR_CLOSED;
929 
930  // cache new directory entry
931  dir = cacheDirEntry(FatCache::CACHE_FOR_WRITE);
932  if (!dir) {
933  DBG_FAIL_MACRO;
934  goto fail;
935  }
936  // copy all but name and name flags to new directory entry
937  memcpy(&dir->creationTimeTenths, &entry.creationTimeTenths,
938  sizeof(entry) - sizeof(dir->name) - 2);
939  dir->attributes = entry.attributes;
940 
941  // update dot dot if directory
942  if (dirCluster) {
943  // get new dot dot
944  uint32_t block = m_vol->clusterFirstBlock(dirCluster);
945  pc = m_vol->cacheFetchData(block, FatCache::CACHE_FOR_READ);
946  if (!pc) {
947  DBG_FAIL_MACRO;
948  goto fail;
949  }
950  memcpy(&entry, &pc->dir[1], sizeof(entry));
951 
952  // free unused cluster
953  if (!m_vol->freeChain(dirCluster)) {
954  DBG_FAIL_MACRO;
955  goto fail;
956  }
957  // store new dot dot
958  block = m_vol->clusterFirstBlock(m_firstCluster);
959  pc = m_vol->cacheFetchData(block, FatCache::CACHE_FOR_WRITE);
960  if (!pc) {
961  DBG_FAIL_MACRO;
962  goto fail;
963  }
964  memcpy(&pc->dir[1], &entry, sizeof(entry));
965  }
966  // Remove old directory entry;
967  oldFile.m_firstCluster = 0;
968  oldFile.m_flags = O_WRITE;
969  oldFile.m_attr = FILE_ATTR_FILE;
970  if (!oldFile.remove()) {
971  DBG_FAIL_MACRO;
972  goto fail;
973  }
974  return m_vol->cacheSync();
975 
976 fail:
977  return false;
978 }

980 bool FatFile::rmdir() {
981  // must be open subdirectory
982  if (!isSubDir() || (!USE_LONG_FILE_NAMES && isLFN())) {
983  DBG_FAIL_MACRO;
984  goto fail;
985  }
986  rewind();
987 
988  // make sure directory is empty
989  while (1) {

991  if (!dir) {
992  // EOF if no error.
993  if (!getError()) {
994  break;
995  }
996  DBG_FAIL_MACRO;
997  goto fail;
998  }
999  // done if past last used entry
1000  if (dir->name[0] == DIR_NAME_FREE) {
1001  break;
1002  }
1003  // skip empty slot, '.' or '..'
1004  if (dir->name[0] == DIR_NAME_DELETED || dir->name[0] == '.') {
1005  continue;
1006  }
1007  // error not empty
1008  if (DIR_IS_FILE_OR_SUBDIR(dir)) {
1009  DBG_FAIL_MACRO;
1010  goto fail;
1011  }
1012  }

1021 //------------------------------------------------------------------------------
1022 bool FatFile::rmRfStar() {
1023  uint16_t index;
1024  FatFile f;
1025  if (!isDir()) {
1026  DBG_FAIL_MACRO;
1027  goto fail;
1028  }
1029  rewind();
1030  while (1) {
1031  // remember position
1032  index = m_curPosition/32;
1033 
1034  dir_t* dir = readDirCache();
1035  if (!dir) {
1036  // At EOF if no error.
1037  if (!getError()) {
1038  break;
1039  }
1040  DBG_FAIL_MACRO;
1041  goto fail;
1042  }
1043  // done if past last entry
1044  if (dir->name[0] == DIR_NAME_FREE) {
1045  break;
1046  }
1047 
1048  // skip empty slot or '.' or '..'
1049  if (dir->name[0] == DIR_NAME_DELETED || dir->name[0] == '.') {
1050  continue;
1051  }
1052 
1053  // skip if part of long file name or volume label in root
1054  if (!DIR_IS_FILE_OR_SUBDIR(dir)) {
1055  continue;
1056  }
1057 
1058  if (!f.open(this, index, O_READ)) {
1059  DBG_FAIL_MACRO;
1060  goto fail;
1061  }
1062  if (f.isSubDir()) {
1063  // recursively delete
1064  if (!f.rmRfStar()) {
1065  DBG_FAIL_MACRO;
1066  goto fail;
1067  }
1068  } else {
1069  // ignore read-only
1070  f.m_flags |= O_WRITE;
1071  if (!f.remove()) {
1072  DBG_FAIL_MACRO;
1073  goto fail;
1074  }
1075  }
1076  // position to next entry if required
1077  if (m_curPosition != (32UL*(index + 1))) {
1078  if (!seekSet(32UL*(index + 1))) {
1079  DBG_FAIL_MACRO;
1080  goto fail;
1081  }
1082  }
1083  }
1084  // don't try to delete root
1085  if (!isRoot()) {
1086  if (!rmdir()) {
1087  DBG_FAIL_MACRO;
1088  goto fail;
1089  }
1090  }
1091  return true;
1092 
1093 fail:
1094  return false;
1095 }
1096 //------------------------------------------------------------------------------
1097 bool FatFile::seekSet(uint32_t pos) {
1098  uint32_t nCur;
1099  uint32_t nNew;
1100  uint32_t tmp = m_curCluster;
1101  // error if file not open
1102  if (!isOpen()) {
1103  DBG_FAIL_MACRO;
1104  goto fail;
1105  }
1106  // Optimize O_APPEND writes.
1107  if (pos == m_curPosition) {
1108  return true;
1109  }
1110  if (pos == 0) {
1111  // set position to start of file
1112  m_curCluster = 0;
1113  goto done;
1114  }
1115  if (isFile()) {
1116  if (pos > m_fileSize) {
1117  DBG_FAIL_MACRO;
1118  goto fail;
1119  }
1120  } else if (isRootFixed()) {
1121  if (pos <= 32*m_vol->rootDirEntryCount()) {
1122  goto done;
1123  }
1124  DBG_FAIL_MACRO;
1125  goto fail;
1126  }
1127  // calculate cluster index for cur and new position
1128  nCur = (m_curPosition - 1) >> (m_vol->clusterSizeShift() + 9);
1129  nNew = (pos - 1) >> (m_vol->clusterSizeShift() + 9);
1130 
1131  if (nNew < nCur || m_curPosition == 0) {
1132  // must follow chain from first cluster
1133  m_curCluster = isRoot32() ? m_vol->rootDirStart() : m_firstCluster;
1134  } else {
1135  // advance from curPosition
1136  nNew -= nCur;
1137  }
1138  while (nNew--) {
1139  if (m_vol->fatGet(m_curCluster, &m_curCluster) <= 0) {
1140  DBG_FAIL_MACRO;
1141  goto fail;
1142  }
1143  }
1144 
1145 done:
1146  m_curPosition = pos;
1147  return true;
1148 
1149 fail:
1150  m_curCluster = tmp;
1151  return false;
1152 }

1348 //------------------------------------------------------------------------------
1349 int FatFile::write(const void* buf, size_t nbyte) {
1350  // convert void* to uint8_t* - must be before goto statements
1351  const uint8_t* src = reinterpret_cast<const uint8_t*>(buf);
1352  cache_t* pc;
1353  uint8_t cacheOption;
1354  // number of bytes left to write - must be before goto statements
1355  size_t nToWrite = nbyte;
1356  size_t n;
1357  // error if not a normal file or is read-only
1358  if (!isFile() || !(m_flags & O_WRITE)) {
1359  DBG_FAIL_MACRO;
1360  goto fail;
1361  }
1362  // seek to end of file if append flag
1363  if ((m_flags & O_APPEND)) {
1364  if (!seekSet(m_fileSize)) {
1365  DBG_FAIL_MACRO;
1366  goto fail;
1367  }
1368  }
1369  // Don't exceed max fileSize.
1370  if (nbyte > (0XFFFFFFFF - m_curPosition)) {
1371  DBG_FAIL_MACRO;
1372  goto fail;
1373  }
1374  while (nToWrite) {
1375  uint8_t blockOfCluster = m_vol->blockOfCluster(m_curPosition);
1376  uint16_t blockOffset = m_curPosition & 0X1FF;
1377  if (blockOfCluster == 0 && blockOffset == 0) {
1378  // start of new cluster
1379  if (m_curCluster != 0) {
1380  int8_t fg = m_vol->fatGet(m_curCluster, &m_curCluster);
1381  if (fg < 0) {
1382  DBG_FAIL_MACRO;
1383  goto fail;
1384  }
1385  if (fg == 0) {
1386  // add cluster if at end of chain
1387  if (!addCluster()) {
1388  DBG_FAIL_MACRO;
1389  goto fail;
1390  }
1391  }
1392  } else {
1393  if (m_firstCluster == 0) {
1394  // allocate first cluster of file
1395  if (!addCluster()) {
1396  DBG_FAIL_MACRO;
1397  goto fail;
1398  }
1399  m_firstCluster = m_curCluster;
1400  } else {
1401  m_curCluster = m_firstCluster;
1402  }
1403  }
1404  }
1405  // block for data write
1406  uint32_t block = m_vol->clusterFirstBlock(m_curCluster) + blockOfCluster;
1407 
1408  if (blockOffset != 0 || nToWrite < 512) {
1409  // partial block - must use cache
1410  // max space in block
1411  n = 512 - blockOffset;
1412  // lesser of space and amount to write
1413  if (n > nToWrite) {
1414  n = nToWrite;
1415  }
1416 
1417  if (blockOffset == 0 && m_curPosition >= m_fileSize) {
1418  // start of new block don't need to read into cache
1419  cacheOption = FatCache::CACHE_RESERVE_FOR_WRITE;
1420  } else {
1421  // rewrite part of block
1422  cacheOption = FatCache::CACHE_FOR_WRITE;
1423  }
1424  pc = m_vol->cacheFetchData(block, cacheOption);
1425  if (!pc) {
1426  DBG_FAIL_MACRO;
1427  goto fail;
1428  }
1429  uint8_t* dst = pc->data + blockOffset;
1430  memcpy(dst, src, n);
1431  if (512 == (n + blockOffset)) {
1432  // Force write if block is full - improves large writes.
1433  if (!m_vol->cacheSyncData()) {
1434  DBG_FAIL_MACRO;
1435  goto fail;
1436  }
1437  }
1438 #if USE_MULTI_BLOCK_IO
1439  } else if (nToWrite >= 1024) {
1440  // use multiple block write command
1441  uint8_t maxBlocks = m_vol->blocksPerCluster() - blockOfCluster;
1442  uint8_t nBlock = nToWrite >> 9;
1443  if (nBlock > maxBlocks) {
1444  nBlock = maxBlocks;
1445  }
1446  n = 512*nBlock;
1447  if (block <= m_vol->cacheBlockNumber()
1448  && block < (m_vol->cacheBlockNumber() + nBlock)) {
1449  // invalidate cache if block is in cache
1450  m_vol->cacheInvalidate();
1451  }
1452  if (!m_vol->writeBlocks(block, src, nBlock)) {
1453  DBG_FAIL_MACRO;
1454  goto fail;
1455  }
1456 #endif // USE_MULTI_BLOCK_IO
1457  } else {
1458  // use single block write command
1459  n = 512;
1460  if (m_vol->cacheBlockNumber() == block) {
1461  m_vol->cacheInvalidate();
1462  }
1463  if (!m_vol->writeBlock(block, src)) {
1464  DBG_FAIL_MACRO;
1465  goto fail;
1466  }
1467  }
1468  m_curPosition += n;
1469  src += n;
