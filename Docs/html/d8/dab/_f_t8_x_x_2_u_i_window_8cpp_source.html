219  uint16_t _height = tft.height();

223  last = pgm_read_byte(&font->last),
224  gw, gh, xa;
225  int x = 0, y = 0;
226  int8_t xo, yo;
227  int16_t minx = _width, miny = _height, maxx = -1, maxy = -1,
228  gx1, gy1, gx2, gy2, ts = tft.getTextSize(),
229  ya = ts * (uint8_t)pgm_read_byte(&font->yAdvance);
230 
231  while ((c = *testStr++)) {
232  if (c != '\n') { // Not a newline
233  if (c != '\r') { // Not a carriage return, is normal char
234  if ((c >= first) && (c <= last)) { // Char present in current font
235  c -= first;
236  glyph = &(((GFXglyph *)pgm_read_pointer(&font->glyph))[c]);
237  gw = pgm_read_byte(&glyph->width);
238  gh = pgm_read_byte(&glyph->height);
239  xa = pgm_read_byte(&glyph->xAdvance);
240  xo = pgm_read_byte(&glyph->xOffset);
241  yo = pgm_read_byte(&glyph->yOffset);
242  gx1 = x + xo * ts;
243  gy1 = y + yo * ts;
244  gx2 = gx1 + gw * ts - 1;
245  gy2 = gy1 + gh * ts - 1;
246  if (gx1 < minx) minx = gx1;
247  if (gy1 < miny) miny = gy1;
248  if (gx2 > maxx) maxx = gx2;
249  if (gy2 > maxy) maxy = gy2;
250  x += xa * ts;
251  }
252  } // Carriage return = do nothing
253  }
254  else { // Newline
255  x = 0; // Reset x
256  y += ya; // Advance y by 1 line
257  }
258  }
