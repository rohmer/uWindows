82 void SdSpiAltDriver::send(const uint8_t* buf , size_t n) {
83  // Adjust to 32-bit alignment.
84  while ((reinterpret_cast<uintptr_t>(buf) & 0X3) && n) {
85  SPI.transfer(*buf++);
86  n--;
87  }
88  SPI.transferBytes(const_cast<uint8_t*>(buf), 0, n);
